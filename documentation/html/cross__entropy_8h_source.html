55 struct CrossEntropy : public LossFunctor {
56  // Factory method.
57  static LossFunctor::Ptr Create() {
58  LossFunctor::Ptr ptr(new CrossEntropy);
59  return ptr;
60  }
61 
62  // All loss functors must evaluate the loss and derivative with respect to
63  // the input 'values' (which are the output of some 'OutputLayer').
64  bool Evaluate(const VectorXd& ground_truth, const VectorXd& values,
65  double& loss, VectorXd& gradient) const {
66  const double kEpsilon = 1e-4;
67 
68  // Check that 'ground truth' and 'values' are probability distributions
69  // on equal-sized alphabets. Also make sure gradient length matches.
70  if (ground_truth.rows() != values.rows() ||
71  gradient.rows() != values.rows()) {
72  LOG(WARNING) << "Ground truth, values, and gradient are not "
73  << "the same length.";
74  return false;
75  }
76 
77  if (ground_truth.sum() < 1.0 - kEpsilon ||
78  ground_truth.sum() > 1.0 + kEpsilon) {
79  LOG(WARNING) << "Ground truth vector does not sum to unity ("
80  << ground_truth.sum() << ").";
81  return false;
82  }
83 
84  if (values.sum() < 1.0 - kEpsilon || values.sum() > 1.0 + kEpsilon) {
85  LOG(WARNING) << "Values vector does not sum to unity (" << values.sum()
86  << ").";
87  return false;
88  }
89 
90  if (ground_truth.minCoeff() < -kEpsilon) {
91  LOG(WARNING) << "Ground truth vector contains a number less than 0.0.";
92  return false;
93  }
94 
95  if (ground_truth.maxCoeff() > 1.0 + kEpsilon) {
96  LOG(WARNING) << "Ground truth vector contains a number greater than 1.0.";
97  return false;
98  }
99 
100  if (values.minCoeff() < -kEpsilon) {
101  LOG(WARNING) << "Values vector contains a number less than 0.0.";
102  return false;
103  }
104 
105  if (values.maxCoeff() > 1.0 + kEpsilon) {
106  LOG(WARNING) << "Values vector contains a number greater than 1.0.";
107  return false;
108  }
109 
110  // Compute the loss and gradient.
111  loss = 0.0;
112  for (size_t ii = 0; ii < ground_truth.rows(); ii++) {
113  loss -= ground_truth(ii) * std::log(std::max(values(ii), kEpsilon));
114 
115  // Catch small values(ii).
116  if (values(ii) < kEpsilon)
117  gradient(ii) = 0.0;
118  else
119  gradient(ii) = -ground_truth(ii) / values(ii);
120  }
121 
122  return true;
123  }
124 }; //\struct CrossEntropy
